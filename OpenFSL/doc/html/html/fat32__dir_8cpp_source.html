14 uint32_t FS_FAT32::getChildCount(std::string path) {
15  std::string currentPath_temp = currentPath;
16  uint32_t currentCluster_temp = currentCluster;
17 
18  if (path != "") {
19  chdir(path);
20  }
21 
22  int count = 0;
23  int countWithDeletedFiles = 0;
24 
25  uint32_t linkedClusterCount = getLinkedClusterCount(currentCluster); // Get linked cluster count.
26  if (linkedClusterCount == 0)
27  {
28  currentPath = currentPath_temp;
29  currentCluster = currentCluster_temp;
30  return 0;
31  }
32 
33  Sector dir_cluster(sectorPerCluster * linkedClusterCount, dd->getBytespersector()); // Get entire linked cluster of directory entry
34 
35  if (getLinkedCluster(&dir_cluster, currentCluster))
36  {
37  currentPath = currentPath_temp;
38  currentCluster = currentCluster_temp;
39  return 0;
40  }
41 
42  FAT32_entry* fileEntry = (FAT32_entry*)dir_cluster.getData();
43 
44  for (uint32_t i = 0; i < 16 * linkedClusterCount; i++) {
45  if (fileEntry[i].fileAttr == 0x00) // return if entry is blank
46  {
47  currentPath = currentPath_temp;
48  currentCluster = currentCluster_temp;
49  return count;
50  }
51  else {
52  if (countWithDeletedFiles < 65536) {
53  countWithDeletedFiles++; // Increase entry count
54  if (fileEntry[i].fileName[0] != 0xE5 && (fileEntry[i].fileAttr == 0x10 || fileEntry[i].fileAttr == 0x20)) // Increase count if file is exist
55  count++;
56  } else { // return if entry count is over 65535
57  currentPath = currentPath_temp;
58  currentCluster = currentCluster_temp;
59  return count;
60  }
61  }
62  }
63 
64  currentPath = currentPath_temp;
65  currentCluster = currentCluster_temp;
66  return count;
67 }

82  }

104 
105  for (uint32_t i = 0; i < 16 * linkedClusterCount; i++) {
106  if (fileEntry[i].fileAttr == 0x00)
107  {
108  currentPath = currentPath_temp;
109  currentCluster = currentCluster_temp;
110  return buf;
111  }
112  else if (fileEntry[i].fileName[0] != 0xE5)
113  {
114  if (fileEntry[i].fileAttr == 0x20 || fileEntry[i].fileAttr == 0x10) {
115  std::string filename;
116 
117  if (fileEntry[i - 1].fileAttr == 0x0F){
118  for (int lfnIndex = i - 1; fileEntry[lfnIndex].fileAttr == 0x0F; lfnIndex--)
119  {
120  char buf[14] = { 0 };
121  FAT32_lfn* entry = (FAT32_lfn*)&fileEntry[lfnIndex];
122  str16to8((uint8_t*)buf, entry->lfnFileName1, 5);
123  str16to8((uint8_t*)(buf + 5), entry->lfnFileName2, 6);
124  str16to8((uint8_t*)(buf + 11), entry->lfnFileName3, 2);
125  filename += buf;
126  }
127  }
128  else {
129  filename = (char*)fileEntry[i].fileName;
130  std::string fileext;
131  if (fileext != " ") {
132  fileext = filename.substr(8, 3);
133  fileext.erase(fileext.find_last_not_of(" ") + 1);
134  }
135  filename = filename.substr(0, 8);
136  filename.erase(filename.find_last_not_of(" ") + 1);
137  if (!fileext.empty())
138  filename += "." + fileext;
139  }
140 
141  buf[index].fileName = filename;
142  buf[index].fileAttr = fileEntry[i].fileAttr;
143  buf[index].fileCreateTime.time_millis = (fileEntry[i].fileCreateTenth % 10) * 100;
144  buf[index].fileCreateTime.time_sec = ((fileEntry[i].fileCreateTime & 0b0000000000011111) * 2) + fileEntry[i].fileCreateTenth / 10;
145  buf[index].fileCreateTime.time_min = (fileEntry[i].fileCreateTime & 0b0000011111100000) >> 5;
146  buf[index].fileCreateTime.time_hour = (fileEntry[i].fileCreateTime & 0b1111100000000000) >> 11;
147  buf[index].fileCreateTime.time_day = (fileEntry[i].fileCreateDate & 0b0000000000011111);
148  buf[index].fileCreateTime.time_month = (fileEntry[i].fileCreateDate & 0b0000000111100000) >> 5;
149  buf[index].fileCreateTime.time_year = ((fileEntry[i].fileCreateDate & 0b1111111000000000) >> 9) + 1980;
150 
151  buf[index].fileModTime.time_sec = (fileEntry[i].fileModTime & 0b0000000000011111) * 2;
152  buf[index].fileModTime.time_min = (fileEntry[i].fileModTime & 0b0000011111100000) >> 5;
153  buf[index].fileModTime.time_hour = (fileEntry[i].fileModTime & 0b1111100000000000) >> 11;
154  buf[index].fileModTime.time_day = (fileEntry[i].fileModDate & 0b0000000000011111);
155  buf[index].fileModTime.time_month = (fileEntry[i].fileModDate & 0b0000000111100000) >> 5;
156  buf[index].fileModTime.time_year = ((fileEntry[i].fileModDate & 0b1111111000000000) >> 9) + 1980;
157 
158  buf[index].fileAccessTime.time_day = (fileEntry[i].fileAccessDate & 0b0000000000011111);
159  buf[index].fileAccessTime.time_month = (fileEntry[i].fileAccessDate & 0b0000000111100000) >> 5;
160  buf[index].fileAccessTime.time_year = ((fileEntry[i].fileAccessDate & 0b1111111000000000) >> 9) + 1980;
161 
162  buf[index].fileSize = fileEntry[i].fileSize;
163  buf[index].fileLocation = (fileEntry[i].fileLocationHigh << 16) + fileEntry[i].fileLocationLow;
164 
165  index++;
166  }
167  }
168  }
169 
170  currentPath = currentPath_temp;
171  currentCluster = currentCluster_temp;
172 
173  return buf;
174 }

273  }
274  }
275  }

277  return 1;
278  }
279  count++;
280  }
